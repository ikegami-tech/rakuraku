/* --- 参加者選択 (タグ方式) ロジック --- */      // 選択されたユーザーIDを保持するSet (重複を防ぐため)   let selectedParticipantIds = new Set();   // 1. 入力があったら候補を表示   function onSearchInput(input) {     const text = input.value.toLowerCase();
